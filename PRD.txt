// lib/utils/marker-parser.ts
export function parseImageMarkers(content: string): {
  segments: ContentSegment[];
  markers: MarkerInfo[];
} {
  const markerRegex = /\[IMAGE_(\d+)\]/g;
  const segments: ContentSegment[] = [];
  const markers: MarkerInfo[] = [];
  
  let lastIndex = 0;
  let match;
  
  while ((match = markerRegex.exec(content)) !== null) {
    // ë§ˆì»¤ ì• í…ìŠ¤íŠ¸
    if (match.index > lastIndex) {
      segments.push({
        type: 'text',
        content: content.substring(lastIndex, match.index),
      });
    }
    
    // ë§ˆì»¤
    const imageIndex = parseInt(match[1]);
    markers.push({
      imageIndex,
      position: match.index,
      markerText: match[0],
    });
    
    segments.push({
      type: 'marker',
      imageIndex,
      markerText: match[0],
    });
    
    lastIndex = match.index + match[0].length;
  }
  
  // ë§ˆì§€ë§‰ í…ìŠ¤íŠ¸
  if (lastIndex < content.length) {
    segments.push({
      type: 'text',
      content: content.substring(lastIndex),
    });
  }
  
  return { segments, markers };
}

// lib/utils/image-guide-generator.ts
export function generateImageGuides(
  content: string,
  images: File[],
  imageAnalyses: CompressedImageAnalysis[]
): ImageGuide[] {
  const { segments, markers } = parseImageMarkers(content);
  const guides: ImageGuide[] = [];
  
  markers.forEach((marker) => {
    const imageIndex = marker.imageIndex - 1; // 0-based
    const imageFile = images[imageIndex];
    const analysis = imageAnalyses[imageIndex];
    
    if (!imageFile || !analysis) return;
    
    // ë¬¸ë§¥ ì¶”ì¶œ
    const { before, after } = extractContext(content, marker.position);
    
    // ì¤„ ë²ˆí˜¸ ê³„ì‚°
    const lineNumber = calculateLineNumber(content, marker.position);
    const paragraphNumber = calculateParagraphNumber(content, marker.position);
    
    guides.push({
      imageIndex: marker.imageIndex,
      imageFile,
      thumbnailUrl: URL.createObjectURL(imageFile),
      markerText: marker.markerText,
      markerPosition: marker.position,
      lineNumber,
      paragraphNumber,
      contextBefore: before,
      contextAfter: after,
      imageDescription: `${analysis.cat} - ${analysis.subj}`,
      recommendedCaption: generateCaption(analysis),
      category: analysis.cat,
    });
  });
  
  return guides;
}

function extractContext(content: string, position: number): {
  before: string;
  after: string;
} {
  const lines = content.split('\n');
  let charCount = 0;
  let targetLine = 0;
  
  for (let i = 0; i < lines.length; i++) {
    charCount += lines[i].length + 1; // +1 for \n
    if (charCount > position) {
      targetLine = i;
      break;
    }
  }
  
  const before = lines.slice(Math.max(0, targetLine - 2), targetLine).join('\n');
  const after = lines.slice(targetLine + 1, Math.min(lines.length, targetLine + 3)).join('\n');
  
  return { before, after };
}

function calculateLineNumber(content: string, position: number): number {
  return content.substring(0, position).split('\n').length;
}

function calculateParagraphNumber(content: string, position: number): number {
  const beforeContent = content.substring(0, position);
  return beforeContent.split('\n\n').length;
}

function generateCaption(analysis: CompressedImageAnalysis): string {
  // ì¹´í…Œê³ ë¦¬ë³„ ìº¡ì…˜ ìƒì„± ë¡œì§

    case 'food':

    case 'product':
      return `${analysis.subj} ìƒì„¸`;
    case 'interior':
      return `${analysis.subj} ê³µê°„`;
    default:
      return analysis.subj;
  }
}
